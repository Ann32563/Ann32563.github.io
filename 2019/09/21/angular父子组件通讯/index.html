angular父子组件通讯

撰写于 2019-09-21 修改于 2019-09-21 分类 angular 标签 angular

父组件

ts:

//主动获取子组件的值

@ViewChild('overview')

overview:OverviewComponent;



// 接收子组件传递过来的数据

runParent(msg:string){

alert(msg);

}

html:

//通过id #overview直接执行子组件的方法，[]属性绑定，传值给子组件，（）事件绑定，获取子组件的事件

<overview #overview [signal]="fathervar" (outer)="runparent($event)" ></overview>

子组件

//接收父组件的值    
@Input() signal;

//传递子组件的事件    
@output outer=new EventEmitter() 
son() { this.outer.emit('hey my father'); }

//监听属性变化    
ngOnChanges(changes: SimpleChanges) {

}

//双向绑定
[(ngModel)]='dubblebind'
